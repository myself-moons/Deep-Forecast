5. Performance Test (Caching Benefits)
# Time a direct prediction vs cached response
python -c "
import time
from predict import run_forecast

# Time direct prediction
start = time.time()
result = run_forecast()
direct_time = time.time() - start

# Time cached load
import json
start = time.time()
with open('model_files/latest_forecast.json') as f:
    cached = json.load(f)
cache_time = time.time() - start

print(f'Direct prediction: {direct_time:.2f}s')
print(f'Cached load: {cache_time:.4f}s')
print(f'✅ Cache is {direct_time/cache_time:.0f}x faster')
"
